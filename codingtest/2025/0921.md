# 입문자를 위한 코테 핵심
## 9회 재귀함수와 DFS

### n! 구하기
cnt = 1
def DFS(n):
    global cnt
    if n == 0:
        return
    else:
        cnt = cnt * n
        DFS(n-1)
        print(cnt, end = ' ')
        

print(DFS(5))             
print(DFS(6))

### 정답
def DFS(n):
    if n == 1:
        return 1
    else:
        return n * DFS(n-1)

print(DFS(5))
print(DFS(6))

### 피보나치 수열
def DFS(n):
    if n == 1:
        return
    else:
        print(n, end=' ')
        DFS(n-1)
              
print(DFS(5))
