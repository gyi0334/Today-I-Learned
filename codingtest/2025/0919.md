## 2. Backspace
def solution(s):
    stack = []
    answer = ""
    for x in s:
        if x == '#':
            stack.pop()
        else:
            stack.append(x)
    answer = ''.join(stack)
    return answer

from collections import deque
def solution(nums):
    queue = deque(nums)
    while queue:
        for _ in range(2):  # 변수 넣기 싫으면 _ 써도 된다.
            if len(queue) >= 2: # 2개만 남았을때는 popleft 안하게 설정
                queue.popleft()
        queue.append(queue.popleft())
        if len(queue) == 1:
            return queue[0]
